import numpy as np
a=np.array([2,6])
b=np.array([3,10])
print("vector:")
print("a:",a)
print("\nb:",b)
print("\n inner product of vector a and b")
print(np.inner(a,b))
print("-------")
print("\n outer product of vector a and b")
print(np.outer(a,b))
print("-------")
print("\n cross product of vectors a and b")
print(np.cross(a,b))
